1//! mPic - Simple Lossy Compression Image Format for Embedded Platforms
2//!
3//! ## Features
4//!
5//! - Simple.
6//! - Lossy compression.
7//! - A typical image compression ratio is somewhere between PNG and JPG.
8//! - Small memory footprint, only a few hundred bytes of stack memory required for decoding.
9//! - Designed for 16bpp color images and supports `embedded-graphics`; add `features = ["embedded"]` to Cargo.toml.
10//! - Support for `no_std`, No `alloc` is needed for decoding.
11//!
12//! ## More information
13//!
14//! - See detail: <https://github.com/neri/mpic>
15//!

85impl FileHeader {
86    pub const MINIMAL_SIZE: usize = size_of::<Self>();
87
88    pub const MAGIC: [u8; 4] = *b"\x00mpi";
89
90    #[inline]
91    pub const fn new(width: u32, height: u32) -> Option<Self> {
92        if width == 0 || width > 0xFFFF || height == 0 || height > 0xFFFF {
93            return None;
94        }
95        let version = if (width & 7) == 0 && (height & 7) == 0 {
96            Version::V0
97        } else {
98            Version::V1
99        };
100        Some(Self {
101            magic: Self::MAGIC,
102            version,
103            width: (width as u16).to_le(),
104            height: (height as u16).to_le(),
105        })
106    }
107
108    #[inline]
109    pub fn is_valid(&self) -> bool {
110        let width = self.width.to_le();
111        let height = self.height.to_le();
112        self.magic == Self::MAGIC
113            && (Version::V0..=Version::CURRENT).contains(&self.version)
114            && width > 0
115            && height > 0
116    }
117
118    #[inline]
119    pub fn from_bytes<'a>(blob: &'a [u8]) -> Option<&'a Self> {
120        if blob.len() < Self::MINIMAL_SIZE {
121            return None;
122        }
123        let header = unsafe { &*(blob.as_ptr() as *const FileHeader) };
124        header.is_valid().then(|| header)
125    }
126
127    #[inline]
128    pub fn bytes<'a>(&'a self) -> &'a [u8] {
129        unsafe { slice::from_raw_parts(self as *const _ as *const u8, size_of::<Self>()) }
130    }
131
132    #[inline]
133    pub fn info(&self) -> ImageInfo {
134        ImageInfo {
135            width: self.width.to_le(),
136            height: self.height.to_le(),
137        }
138    }
139
140    #[inline]
141    pub const fn version(&self) -> Version {
142        self.version
143    }
144}
